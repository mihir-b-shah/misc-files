
int len(char* str) {
	int ct = 0;
	while(!(*str == '\0')) {
		++str;
		++ct;
	}
	return ct;
}

int main() {
    long addr = 0;
    addr = addr-1;
  
    long len = len(str);
    long mid = len*(1L << ((int) addr/len));

    while(mid >= len) {
        if(addr > mid) {
            addr = addr-1-mid;
        } else if(addr == mid) {
            addr = mid-1;
        }
        mid = mid >>> 1;
    }
    return 0;
}
